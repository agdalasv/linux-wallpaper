Instalación

Arch Linux

git clone https://github.com/agdalasv/linux-wallpaper.git
cd linux-wallpaper
makepkg -sri

Ubuntu / Debian

sudo apt install python3-pyqt6 python3-requests python3-pillow
git clone https://github.com/agdalasv/linux-wallpaper.git
cd linux-wallpaper
dpkg-buildpackage -us -uc
sudo dpkg -i ../linux-wallpaper_1.0_all.deb

Fedora

sudo dnf install python3-pyqt6 python3-requests python3-pillow
git clone https://github.com/agdalasv/linux-wallpaper.git
cd linux-wallpaper
sudo cp -r . /opt/linux_wallpaper_app
sudo cp linux-wallpaper-app.sh /usr/bin/
